/*CROSS JOIN*/
SELECT S.STDID, S.STDROLLNO, D.DEPTID, D.DEPTID, D.DEPTNAME
FROM STUDENT S
CROSS JOIN DEPT D

/*LEFT OUTER JOIN*/
SELECT S.STDID,S.STDNAME, S.STDROLLNO, D.DEPTID, D.DEPTNAME
FROM STUDENT S 
LEFT OUTER JOIN DEPT D
ON S.STDID = D.DEPTID


/*RIGHT OUTER JOIN*/
SELECT S.STDID,S.STDNAME, S.STDROLLNO, D.DEPTID, D.DEPTNAME
FROM STUDENT S 
RIGHT OUTER JOIN DEPT D
ON S.STDID = D.DEPTID
